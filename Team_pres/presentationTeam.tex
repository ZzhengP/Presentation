\subsection{Linéaire MPC }
\begin{frame}
\begin{block}{Modèle linéaire discrétisé pour la prédiction}
	\begin{equation}
	x_{k+1}=Ax_k+Bu_k
	\end{equation}
	avec, 
	\begin{align*}
	x_k&=\begin{bmatrix}q_k\\\dot{q}_k\end{bmatrix}\in\mathbb{R}^{2n},\\
	A&=\begin{bmatrix}I&I\Delta t\\0&I\end{bmatrix}\in\mathbb{R}^{2n\times 2n},\\
	B&=\begin{bmatrix}I\frac{\Delta t^2}{2}\\I\Delta t\end{bmatrix}\in\mathbb{R}^{2n\times n}
	\end{align*}
\end{block}

\begin{equation}
\label{mpc_q}
\underbrace{\begin{bmatrix}x_1\\\vdots\\x_N\end{bmatrix}}_{\textbf{x}}=
\underbrace{\begin{bmatrix}A\\\vdots\\A^N\end{bmatrix}}_{P_x}x_0+
\underbrace{\begin{bmatrix}B&0&0\\\vdots&\ddots&0\\A^{N-1}B&\cdots&B\end{bmatrix}}_{P_u}
\underbrace{\begin{bmatrix}u_{0}\\\vdots\\u_{N-1}\end{bmatrix}}_{\textbf{u}}.
\end{equation}
\end{frame}

\begin{frame}
	L'objectif du schéma linéaire du MPC est: 
	 \begin{subequations}
	 	\begin{align}
	 	\mathop{\mbox{minimize}}_{\textbf{u}}\ \ 
	 	&\sum_{k=0}^{N-1}\|x_{k+1}-x_{k+1}^\mathit{des}\|^2_{Q}+\|u_k-u_k^\mathit{des}||^2_{R}\label{eq:sp_final:c}\\  
	 	\mbox{subject to}\ \ 
	 	&\forall\,k\in\{0,\ldots N-1\},\ \underline{u}\leq u_k\leq\overline{u},\\
	 	&\forall\,k\in\{1,\ldots N\},\ \underline{q}\leq q_k\leq\overline{q},\\
	 	&\forall\,k\in\{1,\ldots N-1\},\ \underline{\dot{q}}\leq\dot{q}_k\leq\overline{\dot{q}},\\
	 	&\dot{q}_N=0\label{eq:sp_final:c7}
	 	\end{align}
	 \end{subequations}
 
L'équation (\ref{eq:sp_final:c7})  assure à la fois la faisabilité récurrente et permet d'introduire la notion de "Passive safety" dans notre schéma MPC.

\begin{block}{Passive safety}
	Il est impossible de garantir qu'il y a aucune collision dans un environnement partiel inconnu, mais au moins on peut garantir s'il y a une collision, le robot est à l'arrêt. 
\end{block}
\end{frame}

\subsection{Evitement de la collision en temps continu}
\begin{frame}
\begin{columns}

\column{0.5\textwidth}
\begin{itemize}
	\item Définir contrainte d'évitement en temps continu
\end{itemize}
\begin{figure}
	\centering
\includegraphics[width=0.8\columnwidth, height=0.6\textwidth]{figures/illu_continous_separation.pdf}
\end{figure}

\column{0.5\textwidth}
\begin{itemize}
\item $\{r^1,\,r^2,\ldots\}$  sommets liés à la personne: 
\begin{subequations}\label{eq-ccap}
	\begin{alignat*}{2}
	&a_kp_k^j&&\leq b_k,\\
	&a_kp_{k+1}^j&&\leq b_k,
	\end{alignat*}
\end{subequations}
\item $\{p^1,\,p^2,\ldots\}$  sommets liés au robot:
\begin{subequations}\label{eq-ccar}
	\begin{alignat*}{2}
	&a_kr_k^i&&\geq b_k+d,\\
	&a_kr_{k+1}^i&&\geq b_k+d,
	\end{alignat*}
\end{subequations}
\end{itemize}
\end{columns}
\begin{block}{Plan de séparateur}
	S'il existe un plan $a_{k}x=b_{k}$ qui sépare les sommets de la personne d'un côté et les sommets du robot d'autre côté durant l'intervalle de temps $t_{k}$ et $t_{k+1}$, alors on dit qu'ils ne sont pas en collision pendant cet intervalle de temps. 
\end{block}
\end{frame}

\begin{frame}
	Notre problème de la planification se divise donc en deux sous problèmes: 
	\begin{itemize}
	\item calcule le plan de séparateur: 
			Pour chaque intervalle de temps $[k,\,k+1]$, $k\in\{0,\ldots N-1\}$, on calcule un plan de séparateur avec un LPs (Linear programming) :
		%
		\begin{subequations}\label{eq:lp}
			\begin{align*}
			\mathop{\mbox{maximize}}_{a_k,\,b_k,\,d}\ \ &d\\  
			\mbox{subject to}\ \ 
			&\forall\,j,\ a_kp_k^j\leq b_k,\\
			&\forall\,j,\ a_kp_{k+1}^j\leq b_k,\\
			&\forall\,i,\ a_kr_k^i\geq b_k+d,\\
			&\forall\,i,\ a_kr_{k+1}^i\geq b_k+d,\\
			&-1\leq a_k\leq1,\\
			&1-\varepsilon\leq a_k^p\,a_k^T\leq1
			\end{align*}
		\end{subequations}
	\end{itemize}
	\end{frame}


\begin{frame}
\begin{itemize}

	\item calcule la trajectoire sans collision avec les plans calculés précédemment: 
	\begin{subequations}\label{eq:qp}
		\begin{align*}
		\mathop{\mbox{min.}}_{\textbf{u}}\ \ 
		&\sum_{k=0}^{N-1}\|x_{k+1}-x_{k+1}^\mathit{des}\|^2_{Q}+\|u_k-u_k^\mathit{des}||^2_{R}\\  
		\mbox{s.t.}\ \ 
		&\forall\,k\in\{0,\ldots N-1\},\ \underline{u}\leq u_k\leq\overline{u},\\
		&\forall\,k\in\{1,\ldots N\},\ \underline{q}\leq q_k\leq\overline{q},\\
		&\forall\,k\in\{1,\ldots N-1\},\ \underline{\dot{q}}\leq\dot{q}_k\leq\overline{\dot{q}},\\
		&\dot{q}_N=0,\\
		&\forall\,k\in\{0,\ldots N-1\},\ \forall\,i,\nonumber\\
		&\ \ \ \ \ a_kr^i\left(q_k^p\right)+a_kJ\left(q_k^p\right)\left(q_k-q_k^p\right)\geq b_k+d_\mathit{safe},\\
		&\forall\,k\in\{0,\ldots N-1\},\ \forall\,i,\nonumber\\
		& a_kr^i\left(q_{k+1}^p\right)+a_kJ\left(q_{k+1}^p\right)\left(q_{k+1}-q_{k+1}^p\right)\geq b_k+d_\mathit{safe}
		\end{align*}
	\end{subequations}
	\end{itemize}
\end{frame}

\subsection{Choix de l'horizon de la prédiction}
\begin{frame}
\begin{block}{borne inférieur}
	Le robot doit être en arrêt à la fin de l'horizon de la prédiction  $\rightarrow$ T$_{low}$ = $\frac{V_{max}}{a_{max}}$ 
\end{block}
Prenons $V_{max}$ = 2~m.s$^{-1}$ et $a_{max}$ = 10~m.s$^{-2}$ , T$_{low}$ = 0.2~s. 

\begin{block}{borne supérieur}
	La vitesse de la main de l'humain en situation régulière est 1~m.s$^{-1}$  avec une accélération de 5~m.s$^{-2}$. 
\end{block}
Considérons $d_{safe}$ = 20~cm,  T$_{upper}$ = 0.3~s. La prédiction du mouvement de la personne peut être pas du tout précis si on prédit trop. 
\end{frame}
